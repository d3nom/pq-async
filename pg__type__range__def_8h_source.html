31 enum class range_flag
32 {
33  none = 0x00,
34  empty = 0x01, /* range is empty */
35  lb_inc = 0x02, /* lower bound is inclusive */
36  ub_inc = 0x04, /* upper bound is inclusive */
37  lb_inf = 0x08, /* lower bound is -infinity */
38  ub_inf = 0x10, /* upper bound is +infinity */
39  lb_null = 0x20, /* lower bound is null (NOT USED) */
40  ub_null = 0x40, /* upper bound is null (NOT USED) */
41  contain_empty = 0x80 /* marks a GiST internal-page entry whose
42  * subtree contains some empty ranges
43  */
44 };

211  inline T lbound_abs() const
212  {
213  if(!this->has_lbound())
214  throw pq_async::exception("lbound is unbounded");

226  return _ub;

340  const fwd_iterator &operator ++() // pre-increment
341  {

343  throw pq_async::exception("Out of bounds iterator increment!");
344  ++_v;
345  return *this;

353  fwd_iterator copy(*this);
