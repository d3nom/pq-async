165  sp_connection_lock open_connection(
166  int32_t timeout_ms = PQ_ASYNC_DEFAULT_TIMEOUT_MS)
167  {
168  if(_lock)
169  return _lock;
170 
171  #ifdef PQ_ASYNC_THREAD_SAFE
172  std::unique_lock<std::recursive_mutex> lock(
173  connection_pool::instance()->conn_pool_mutex
174  );
175  #endif
176 
177  if(_conn == NULL)
178  _conn = connection_pool::get_connection(
179  this, _connection_string, timeout_ms
180  );
181 
182  _conn->reserve();
183  if(_conn == NULL)
184  _conn = connection_pool::get_connection(
185  this, _connection_string, timeout_ms
186  );
187 
188  sp_connection_lock cl(new connection_lock(_conn));
189  _conn->open_connection();
190 
191  #ifdef PQ_ASYNC_THREAD_SAFE
192  lock.unlock();
193  connection_pool::notify_all();
194  #endif
195 
196  return cl;
197  }

207  template<
208  typename CB,
209  PQ_ASYNC_VALID_DB_VAL_CALLBACK(CB, sp_connection_lock)
210  >
211  void open_connection(const CB& acb,
212  int32_t timeout_ms = PQ_ASYNC_DEFAULT_TIMEOUT_MS)
213  {
214  value_cb<sp_connection_lock> cb;
215  assign_value_cb<value_cb<sp_connection_lock>, sp_connection_lock>(
216  cb, acb
217  );
218 
219  if(_lock){
220  this->_strand->push_back(
221  std::bind(cb, nullptr, _lock)
222  );
223  return;
224  }
225 
226  try{
227  auto ct = std::make_shared<connection_task>(
228  this->_strand.get(), this->shared_from_this(), this->_conn,
229  [self=this->shared_from_this(), cb](
230  const cb_error& err, sp_connection_lock lock
231  )-> void {
232  if(err){
233  cb(err, sp_connection_lock());
234  return;
235  }
236 
237  try{
238  cb(nullptr, lock);
239  }catch(const std::exception& err){
240  cb(pq_async::cb_error(err), sp_connection_lock());
241  }
242  });
243  pq_async_log_debug(
244  "queuing open connection\ndb: %x, ct: %x",
245  this, ct.get()
246  );
247 
248  ct->connect(_connection_string, timeout_ms);
249  this->_strand->push_back(ct);
250 
251  }catch(const std::exception& err){
252  cb(pq_async::cb_error(err), sp_connection_lock());
253  }

323  template< typename CB, PQ_ASYNC_VALID_DB_ASY_CALLBACK(CB)>
324  void begin(const CB& acb)
325  {
326  async_cb cb;
327  assign_async_cb<async_cb>(cb, acb);
328  if(this->in_transaction()){
329  this->_strand->push_back(
330  std::bind(cb, cb_error("Already in a transaction!"))
331  );
332  return;
333  }
334 
335  this->open_connection(
336  [self=this->shared_from_this(), cb]
337  (const cb_error& err, sp_connection_lock lock){
338  if(err){
339  cb(err);
340  return;
341  }
342  self->_lock = lock;
343 
344  self->execute("BEGIN",
345  [self, cb](const cb_error& err){
346  if(err){
347  self->_lock.reset();
348  cb(err);
349  return;
350  }
351 
352  self->_conn->is_in_transaction.store(true);
353  cb(nullptr);
354  });
355  });
356  }
