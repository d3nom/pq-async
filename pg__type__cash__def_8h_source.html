246  {
247  int64_t m = std::pow(10L, (int64_t)frac_digits);
248  int64_t n = _val / m;

311  static money from_numeric(const numeric& b, int frac_digits)
312  {
313  int64_t m = std::pow(10L, (int64_t)frac_digits);
314 
315  //_val = b._val;
316  std::string bs = b;
317  money r;
318 
319  std::size_t dp = bs.find(".");
320  if(dp == std::string::npos){
321  r._val = (int64_t)b * m;
322  } else {
323  std::string nbs = bs.substr(0, dp);
324  std::string dbs = bs.substr(dp +1);
325  if(dbs.size() > (size_t)frac_digits)
326  dbs = dbs.substr(0, frac_digits);
327  else if(dbs.size() < (size_t)frac_digits)
328  dbs.append((size_t)frac_digits - dbs.size(), '0');
329 
330  r._val = str_to_num<int64_t>(nbs) * m;
331  r._val += str_to_num<int64_t>(dbs);
332  }
333 
334  return r;
335  }
