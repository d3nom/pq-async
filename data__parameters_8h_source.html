364 template <class T, int32_t dim_count>
365 boost::multi_array<T, dim_count> pgval_to_array(char* val, int len, int fmt)
366 {
367  if(!fmt)
368  throw pq_async::exception(PQ_ASYNC_ERR_NO_TEXT_FMT);
369 
370  boost::array<int32_t, dim_count> dimSizes;
371 
372  // dimension count.
373  int32_t svrDimCount = 0;
374  pq_async::swap4((int32_t*)val, &svrDimCount, false);
375  val += 4;
376 
377  if(svrDimCount != dim_count)
378  throw pq_async::exception(PQ_ASYNC_ERR_DIM_NO_MATCH);
379 
380  // skip NULL flag.
381  val += 4;
382 
383  // verify element OID.
384  int32_t eleOid = 0;
385  pq_async::swap4((int32_t*)val, &eleOid, false);
386  val += 4;
387 
388  //TODO: verify eleOid vs T.
389 
390  // fetch dim size.
391  int32_t eleCount = 1;
392  for(int32_t i = 0; i < svrDimCount; ++i){
393  int32_t dimVal = 0;
394  pq_async::swap4((int32_t*)val, &dimVal, false);
395 
396  eleCount *= dimVal;
397 
398  dimSizes[i] = dimVal;
399  val += 4;
400 
401  //TODO: maby check what to do with the dimension lower bound.
402  val += 4;
403  }
404 
405  boost::multi_array<T, dim_count> result_array(dimSizes);
406 
407  std::vector<T> values;
408 
409  for(int32_t i = 0; i < eleCount; ++i){
410  int32_t valLen = 0;
411  pq_async::swap4((int32_t*)val, &valLen, false);
412  val += 4;
413 
414  T value = val_from_pgparam<T>(eleOid, val, valLen, 1);
415  val += valLen;
416 
417  values.push_back(value);
418  }
419 
420  result_array.assign(values.begin(), values.end());
421 
422  return result_array;
423 }

487 template< class T, int32_t dim_count>
488 pq_async::parameter* new_parameter_gen(
489  boost::multi_array<T, dim_count> value,
490  int32_t eleOid, int32_t arrEleOid
491  )
492 {
493  std::vector<char> membuf2;
494 
495  // dimension count.
496  int32_t dimCount = dim_count;
497  int32_t outDimCount = 0;
498  pq_async::swap4(&dimCount, &outDimCount, true);
499  vec_append(membuf2, (char*)&outDimCount, sizeof(outDimCount));
500 
501  // null flag.
502  int32_t nullFlag = 0;
503  int32_t outNullFlag = 0;
504  pq_async::swap4(&nullFlag, &outNullFlag, true);
505  vec_append(membuf2, (char*)&outNullFlag, sizeof(outNullFlag));
506 
507  int32_t outEleOid = 0;
508  pq_async::swap4(&eleOid, &outEleOid, true);
509  vec_append(membuf2, (char*)&outEleOid, sizeof(outEleOid));
510 
511  // dimension sizes.
512  for(int32_t i = 0; i < dim_count; ++i){
513  // dim size.
514  int32_t dimEleCnt = value.shape()[i];
515  int32_t outDimEleCnt = 0;
516  pq_async::swap4(&dimEleCnt, &outDimEleCnt, true);
517  vec_append(membuf2, (char*)&outDimEleCnt, sizeof(outDimEleCnt));
518 
519  // lbound.
520  int32_t lBound = 0;
521  int32_t outLBound = 0;
522  pq_async::swap4(&lBound, &outLBound, true);
523  vec_append(membuf2, (char*)&outLBound, sizeof(outLBound));
524  }
525 
526  T* itVal = value.data();
527 
528  int32_t num_elements = (int32_t)value.num_elements();
529  for(int32_t i = 0; i < num_elements; ++i){
530  pq_async::parameter* param =
531  pq_async::new_parameter(*((const T*)itVal));
532 
533  char* val = (char*)param->get_value();
534  int32_t len = param->get_length();
535 
536  int32_t outLen = 0;
537  pq_async::swap4(&len, &outLen, true);
538  vec_append(membuf2, (char*)&outLen, sizeof(int32_t));
539 
540  vec_append(membuf2, val, len);
541 
542  delete param;
543  ++itVal;
544  }
545 
546  int size = membuf2.size();
547 
548  char* buf = new char[size];
549  for(int32_t i = 0; i < size; ++i)
550  buf[i] = membuf2[i];
551 
552  return new pq_async::parameter(arrEleOid, buf, size, 1);
553 }
