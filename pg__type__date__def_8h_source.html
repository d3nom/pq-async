61  operator hhdate::sys_time<std::chrono::microseconds>() const
62  {
63  int32_t y = this->_m / 12;
64  int32_t m = this->_m % 12;
65 
66  hhdate::sys_time<std::chrono::microseconds> st(
67  this->_t.to_duration()
68  );
69 
70  auto sd = hhdate::floor<hhdate::days>(st);
71  auto tod = st - sd;
72 
73  hhdate::year_month_day ymd = sd;
74  if(y > 0)
75  ymd += hhdate::years{y};
76  if(m > 0)
77  ymd += hhdate::months{m};
78 
79  // if(!ymd.ok())
80  // ymd = ymd.year()/ymd.month()/ymd.day();
81  return hhdate::sys_days{ymd} + tod;
82  }

184  explicit timestamp(const timestamp_tz& tz);

186  int64_t pgticks() const
187  {
188  auto daypoint = hhdate::floor<hhdate::days>(this->_ts);
189  // calendar date
190  auto ymd = hhdate::year_month_day(daypoint);
191 
192  if(ymd.year() == hhdate::year(0))
193  throw pq_async::exception(
194  "Postgresql doesn't support year zero!"
195  );
196 
197  if(ymd.year() > hhdate::year(0))
198  return _ts.time_since_epoch().count();
199 
200  // hhdate to pg alignment...
201  auto tod = hhdate::make_time(_ts - daypoint);
202  ymd += hhdate::years{1};
203  return ((hhdate::sys_time<std::chrono::microseconds>)
204  hhdate::sys_days(ymd) +
205  tod.hours() + tod.minutes() +
206  tod.seconds() + tod.subseconds()
207  ).time_since_epoch().count();
208  }
209 
210  std::string iso_string() const;
211  operator std::string() const { return iso_string();}
212 
213  template< typename Dur >
214  operator hhdate::sys_time<Dur>() const
215  {
216  return hhdate::floor< Dur >(this->_ts);
