242 template<
243  size_t size,
244  size_t align,
245  typename R,
246  typename... Args
247 > class inplace
248 {
249 public:
250  using storage_t = std::aligned_storage_t<size, align>;
251 
252  using invoke_ptr_t = R(*)(storage_t&, Args&&...);
253  using copy_ptr_t = void(*)(storage_t&, storage_t&);
254  using destructor_ptr_t = void(*)(storage_t&);
255 
256  explicit inplace() noexcept :
257  invoke_ptr_{ detail::empty_inplace<R, storage_t, Args...> },
258  copy_ptr_{ copy_op<std::nullptr_t, storage_t>() },
259  destructor_ptr_{ nullptr }
260  {}
261 
262  template<
263  typename T,
264  typename C = typename detail::closure_decay<T, R, Args...>::type
265  > explicit inplace(T&& closure) noexcept :
266  invoke_ptr_{ static_cast<invoke_ptr_t>(
267  [](storage_t& storage, Args&&... args) -> R
268  { return reinterpret_cast<C&>(storage)(std::forward<Args>(args)...); }
269  ) },
270  copy_ptr_{ copy_op<C, storage_t>() },
271  destructor_ptr_{ static_cast<destructor_ptr_t>(
272  [](storage_t& storage) noexcept -> void
273  { reinterpret_cast<C&>(storage).~C(); }
274  ) }
275  {
276  static_assert(sizeof(C) <= size,
277  "inplace delegate closure too large");
278 
279  static_assert(std::alignment_of<C>::value <= align,
280  "inplace delegate closure alignment too large");
281 
282  new(&storage_)C{ std::forward<T>(closure) };
283  }
284 
285  inplace(const inplace& other) :
286  invoke_ptr_{ other.invoke_ptr_ },
287  copy_ptr_{ other.copy_ptr_ },
288  destructor_ptr_{ other.destructor_ptr_ }
289  {
290  copy_ptr_(storage_, other.storage_);
291  }
292 
293  inplace(inplace&& other) :
294  storage_ { std::move(other.storage_) },
295  invoke_ptr_{ other.invoke_ptr_ },
296  copy_ptr_{ other.copy_ptr_ },
297  destructor_ptr_{ other.destructor_ptr_ }
298  {
299  other.destructor_ptr_ = [](storage_t&) -> void {};
300  }
301 
302  inplace& operator= (const inplace& other)
303  {
304  if (this != std::addressof(other))
305  {
306  invoke_ptr_ = other.invoke_ptr_;
307  copy_ptr_ = other.copy_ptr_;
308 
309  if (destructor_ptr_)
310  destructor_ptr_(storage_);
311 
312  copy_ptr_(storage_, other.storage_);
313  destructor_ptr_ = other.destructor_ptr_;
314  }
315  return *this;
316  }
317 
318  inplace& operator= (inplace&& other)
319  {
320  if (this != std::addressof(other))
321  {
322  if (destructor_ptr_)
323  destructor_ptr_(storage_);
324 
325  storage_ = std::move(other.storage_);
326 
327  invoke_ptr_ = other.invoke_ptr_;
328  copy_ptr_ = other.copy_ptr_;
329  destructor_ptr_ = other.destructor_ptr_;
330 
331  other.destructor_ptr_ = [](storage_t&) -> void {};
332  }
333  return *this;
334  }
335 
336  ~inplace()
337  {
338  if (destructor_ptr_)
339  destructor_ptr_(storage_);
340  }
341 
342  R operator() (Args&&... args) const
343  {
344  return invoke_ptr_(storage_, std::forward<Args>(args)...);
345  }
346 
347  bool empty() const noexcept
348  {
349  return destructor_ptr_ == nullptr;
350  }
351 
352  template<typename T> T* target() const noexcept
353  {
354  return reinterpret_cast<T*>(&storage_);
355  }
356 
357 private:
358  mutable storage_t storage_;
359 
360  invoke_ptr_t invoke_ptr_;
361  copy_ptr_t copy_ptr_;
362  destructor_ptr_t destructor_ptr_;
363 
364  template<
365  typename T,
366  typename S,
367  typename std::enable_if_t<
368  std::is_copy_constructible<T>::value, int
369  > = 0
370  > copy_ptr_t copy_op()
371  {
372  return [](S& dst, S& src) noexcept -> void
373  {
374  new(&dst)T{ reinterpret_cast<T&>(src) };
375  };
376  }
377 
378  template<
379  typename T,
380  typename S,
381  typename std::enable_if_t<
382  !std::is_copy_constructible<T>::value, int
383  > = 0
384  > copy_ptr_t copy_op()
385  {
386  static_assert(std::is_copy_constructible<T>::value,
387  "constructing delegate with move only type is invalid!");
388  }
