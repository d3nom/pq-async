228  double bv = (b._x * b._x + b._y * b._y);

260  {

265

355  static bool intersect(const lseg& a, const lseg& b)
356  {
357  point p = a.a();
358  point r = a.b() - a.a();
359  point q = b.a();
360  point s = b.b() - b.a();
361 
362  double t = point::cross((q - p), s) / point::cross(r, s);
363  double u = point::cross((q - p), r) / point::cross(r, s);
364  return (0.0 <= t && t <= 1.0) && (0.0 <= u && u <= 1.0);
365  }
366 
367  static point get_intersect(const lseg& a, const lseg& b)
368  {
369  point aa = a.a();
370  point ab = a.b();
371  point ba = b.a();
372  point bb = b.b();
373 
374  double px = (aa._x*ab._y - aa._y*ab._x)*(ba._x - bb._x) -
375  (ba._x*bb._y - ba._y*bb._x)*(aa._x - ab._x);
376  double py = (aa._x*ab._y - aa._y*ab._x)*(ba._y - bb._y) -
377  (ba._x*bb._y - ba._y*bb._x)*(aa._y - ab._y);
378  double denominator = (aa._x - ab._x)*(ba._y - bb._y) -
379  (aa._y - ab._y)*(ba._x - bb._x);
380 
381  return point(px / denominator, py / denominator);
382  }
383 
384 private:
385  point _p[2];
