317  virtual void run(uint32_t usec_wait = 1)
318  {
319  do{
320  #ifdef PQ_ASYNC_THREAD_SAFE
321  std::unique_lock<std::mutex> lock(_mutex);

323  lock.unlock();
324  for(size_t i = 0; i < tmp_tasks.size(); ++i)
325  run_event_task(tmp_tasks[i]);

328  lock.unlock();

367  void run_event_task(sp_event_task& t)
368  {
369  if(!t)
370  return;
371 
372  t->run_task();
373  event_requeue_pos pos = t->requeue();
374  if(pos == event_requeue_pos::none)
375  return;
376 
377  if(pos == event_requeue_pos::back){
378  PQ_ASYNC_LOCK_EVENT_QUEUE;
379  _tasks.emplace_back( t );
380  }
381  if(pos == event_requeue_pos::front){
382  PQ_ASYNC_LOCK_EVENT_QUEUE;
383  _tasks.emplace_front( t );
384  }
385  }
