121 // trim from start (in place)

123 {
124  s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int ch) {
125  return !std::isspace(ch);

129 // trim from end (in place)

131 {
132  s.erase(std::find_if(s.rbegin(), s.rend(), [](int ch) {
133  return !std::isspace(ch);

167 {
